import java.io.BufferedWriter;
import java.io.FileWriter;
import java.io.IOException;
import java.util.Scanner;

public class LoginStorage {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Nhập username: ");
        String username = scanner.nextLine();

        System.out.print("Nhập password: ");
        String password = scanner.nextLine();

        // Ghi thông tin vào file login.txt
        try (BufferedWriter writer = new BufferedWriter(new FileWriter("login.txt", true))) {
            writer.write("Username: " + username + ", Password: " + password);
            writer.newLine(); // Thêm dòng mới sau mỗi thông tin
            System.out.println("Thông tin đã được lưu vào file login.txt");
        } catch (IOException e) {
            System.out.println("Có lỗi xảy ra khi ghi vào file: " + e.getMessage());
        }

        scanner.close();
    }
}
